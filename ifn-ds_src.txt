function buildModalContent(lang) {
    const t = translations[lang] || translations.en;
    const createSection = (title, features) => {
        if (!title || !features || features.length === 0) return '';
        const listItems = features.map(item => `<li class="flex items-start"><span class="text-blue-400 mr-3 mt-1">âœ“</span><span>${item}</span></li>`).join('');
        return `
            <div>
                <h3 class="text-xl font-semibold mb-3 text-white">${title}</h3>
                <ul class="space-y-2 list-none pl-0">${listItems}</ul>
            </div>
        `;
    };

    return `
        ${createSection(t.cpanelTitle, t.cpanelFeatures)}
        ${createSection(t.emailTitle, t.emailFeatures)}
        ${createSection(t.codeTitle, t.codeFeatures)}
        ${createSection(t.metricsTitle, t.metricsFeatures)}
    `;
}


function updateContent(lang) {
    const t = translations[lang] || translations.en;
    document.getElementById("language-label-text").textContent = t.labelText;
    document.getElementById("main-title").textContent = t.title;
    document.getElementById("subtitle").textContent = t.subtitle;
    document.getElementById("description").textContent = t.description;
    document.getElementById("upgrade-text").textContent = t.upgradeText;
    document.getElementById("pricing-text").textContent = t.pricingText;
    document.getElementById("upgrade-link").textContent = t.linkText;
    document.getElementById("view-more-btn").textContent = t.viewMoreBtn;
    document.getElementById("copyright-text").textContent = t.copyrightText;
    
    const benefitsList = document.getElementById("benefits-list");
    if (benefitsList) {
        benefitsList.innerHTML = t.benefits.map(benefit => `<li>${benefit}</li>`).join('');
    }

    // Modal Content
    document.getElementById("modal-title-text").textContent = t.modalTitle;
    document.getElementById("modal-content").innerHTML = buildModalContent(lang);

    document.documentElement.lang = lang;
    document.title = t.title;
    const langSelector = document.getElementById("language-selector");
    langSelector.value = lang;
}

function populateLanguageSelector() {
    const selector = document.getElementById("language-selector");
    selector.innerHTML = '';
    
    const autoOption = document.createElement('option');
    autoOption.value = "auto";
    autoOption.textContent = "Auto-detect";
    selector.appendChild(autoOption);

    Object.keys(translations).sort((a, b) => translations[a].langName.localeCompare(translations[b].langName)).forEach(key => {
        const data = translations[key];
        const option = document.createElement('option');
        option.value = key;
        option.textContent = data.langName;
        selector.appendChild(option);
    });
}

function getCurrentDomain() {
    const hostParam = new URLSearchParams(window.location.search).get('host');
    if (hostParam && hostParam.trim() !== '') {
        return hostParam;
    }
    return window.location.hostname;
}

function parseDomain(domain) {
    let cleanDomain = domain.replace(/^https?:\/\//, '');
    const parts = cleanDomain.split('.');
    if (parts.length >= 2) {
        const sld = parts[0];
        const tld = parts.slice(1).join('.');
        return { sld, tld };
    }
    return { sld: cleanDomain, tld: '' };
}

function getBrowserLanguage() {
    const lang = navigator.language || navigator.userLanguage;
    return lang ? lang.split('-')[0] : 'en';
}
